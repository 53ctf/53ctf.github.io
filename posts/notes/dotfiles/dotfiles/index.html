Portscan.sh

./portscan.sh IP_ADDRESS

#!/bin/bash

mkdir -p nmap

echo
nmap -Pn -p- -T4 -oN nmap/all_ports $1 -v | grep Discovered
echo
nmap -Pn -p "$(cat nmap/all_ports | grep open | awk -F '/' '{print $1}' ORS=',')" -A -T4 -oN nmap/aggresive $1
sudo nmap -Pn -A -T5 -oN nmap/udp -sU $1

htb.sh

#!/bin/bash

mkdir /home/syclops/htb/machines/$1
cd /home/syclops/htb/machines/$1

tmux new-session -s htb -d -n VPN
tmux new-window -t htb -d -n ATK
tmux new-window -t htb -d -n SVR
tmux new-window -t htb -d -n NTS

tmux select-window -t htb:VPN
tmux send-keys -t htb:VPN "echo 5369| sudo -S /usr/sbin/openvpn /home/syclops/vpn/lab_SYCLOPS.ovpn" Enter


tmux select-window -t htb:SVR
tmux send-keys "cd /home/syclops/tools; clear; echo 5369| sudo -S /usr/bin/python3 -m http.server 80" Enter

tmux select-window -t htb:NTS
tmux send-keys -t htb:NTS "/usr/bin/vim notes.md" Enter

tmux select-window -t htb:ATK
tmux attach -t htb
